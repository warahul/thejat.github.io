R for Machine Learning

Date Thu 29 May 2014 Tags ml

A short recap for myself about some data mining related functionalities in R. This is from Allison Chang's note on 'R for Machine Learning' for MIT 15.097 Spring 2012.

Absolute 0

#path-to-directory, library-name and function-name are strings to be replaced below.
getwd() #get working directory
setwd("path-to-directory") #set working directory to your project folder
library(library-name) #load a particular library, for example e1071 for SVMs
?function-name #help

Datasets

data() #R's own datasets
data(trees) #load the trees dataset which is a 31*3 'data frame' (== matrix with entries possibly strings/non-numerical) 
str(trees) #trees is a bit different from a matrix with column names. This command should return 'data.frame' and other info

Vectors

Everything in R is an object and the assignment operator <- is used to store the object into variables.

v <- 99 # variable v is 'assigned' value 99. Scalars are vectors of length 1.
v <- c(5,6,7,8) #variable v is now a vector. c is a vector concat operator
v <- t(v) #transpose operation. Check how the rownames and columnnames look like
v <- seq(from=100,to=90,by=-2) #handy function to generate the vector [100 98 96 94 92 90]
v <- seq(from=1, to=1, length.out=3) #all ones vector
v <- rep(c("theja","r"),c(3,2)) #elements of the vector can be characters, strings, T/F boolean. rep is for repetition
v[1] #indexing
v[c(1,4,5)] #indexing
v[-c(1,4,5)] #negative indexing
v[v<4] #conditional indexing
which(v<4) #indices of elements satisfying the criteria
length(v) #length

Matrices

v1 <- c(1,2,3)
v2 <- c(4,5,6)
mat1 <- cbind(v1,v2) #concatenates the two vectors side-by-side, as two columns of a matrix
rbind(v1,v2) #concatenates the two vectors on top of each other as two rows of a matrix
mat2 <- matrix(seq(from=1,to=6,by=1),nrow=3,ncol=2,byrow=TRUE) #same as the first three commands. Except the columns names are not v1, v2 anymore.
colnames(mat2) <- c("v1","v2") #not this is the same as the first three commands in this blurb

mat2[,"v1"] #pick the first column. This gives a vector.
mat2[,1] #same as above.
mat2[2,] #pick the second row.
nrow(mat2) #number of rows
ncol(mat2) #number of columns

I/O

#path-to-file is the absolute or relative path to a csv/tsv like file you want to read 
mydata <- read.table("path-to-file", header=TRUE, sep=",") #sep="," means we want to read a csv file. sep="\t" means a tsv file
write.table(...) #try ?write.table
cat(paste("theja","may know R.")) #printing to output. paste is to concatenate strings with a whitespace between them.
as.data.frame(matrix-name) #to transform into a data frame
as.factor(vector-name) #to transform into a factor variable. It is a vector of categorical variables encoded using integers.

Sampling

rnorm(n=2,mean=0,sd=1) #a standard Gaussian vector of length 2
rexp(n=2,rate=3) #an exponential random vector of rate 3 (check the definition of rate using ?rexp . Believe me, this will save you some trouble later on.)
rpois(n=2,lambda=1) #a Poisson random vector (again check the definition of lambda using ?rpois)
runif(n=2,min=1,max=3) #uniform
rbinom(n=2,size=10,prob=.5) #flipping an unbiased coin 10 times. This is related 2 times and the number of heads is output.
sample(c(2,3,4,5,6),size=100,replace=TRUE) #sampling from a vector of numbers 100 times uniformly with replacement

Loop and conditioning

for (condition){
    #do something
    if(another-condition){
        #do something additional
    }
}

Plotting

plot(mydata[,1],mydata[,2],main="Scatterplot of x vs y") #scatterplot
hist(mydata[,1]) #histogram
boxplot(v2~v1,as.data.frame(mat2)) #boxplot: v1 and v2 are the two column names of matrix mat2. The first argument is a formula.

Formula are of the form y ~ x1 + x2 + x3 . Check with ?formula

Data models

#Linear regression
lm_model <- lm(y ~ x1 + x2, data = as.data.frame(cbind(y,x1,x2)))
summary(lm_model)
lm_model$coefficients
predicted <- predict.lm(lm_model,newdata=as.data.frame(cbind(x1_test,x2_test)))

#Logistic regression for classification
glm_mod <- glm(y ~ x1 + x2, family=binomial(link="logit"), data=as.data.frame(cbind(y,x1,x2)))

#K-NN classification using the class package
library("class")
knn_model <- knn(train=cbind(x1,x2), test=cbind(x1_test,x2_test),cl=as.factor(y),k=K)

#Naive Bayes classifier from e1071 package
library("e1071")
NB_model <- naiveBayes(y ~x1 + x2, data= as.data.frame(cbind(y,x1,x2)))

#Decision trees / CART from rpart package for classification
library("rpart")
cart_model <- rpart(y ~ x1 + x2, data=as.data.frame(cbind(y,x1,x2)), method="class") #use plot.rpart and text.rpart as needed

#Adaboost in ada package using decision trees from the rpart package
library("ada")
boost_model <- ada(x=cbind(x1,x2),y=y)

#SVMs from e1071 package
library("e1071")
svm_model <- svm(x = cbind(x1,x2), y=as.factor(y), kernel="radial", cost=C)
summary(svm_model)

#K-means Clustering
kmeans_model <- kmeans(x=cbind(x1,x2), centers = m) #here m is the number of clusters
kmeans(x=c(rnorm(n=10,mean=0,sd=1),rnorm(n=10,mean=10,sd=1)),centers=2) #for example

#Apriori from arules package runs on a binary incidence matrix: column names are items and rows are transactions
library("arules")
generated_rules <- apriori(mydata, parameter=list(supp=0.8, conf=0.9))
summary(generated_rules)
inspect(generated_rules)
